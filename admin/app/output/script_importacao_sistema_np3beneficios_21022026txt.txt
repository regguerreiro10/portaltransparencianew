(
  idold,

  descricaopedido,
  dt_pedido,
  mes,
  ano,
  km,
  system_unit_id,
  entidade_id,

SELECT
  distinct p.id_pedido                                           AS idold,
  vei.id                                                AS veiculos_id,
  p.servico                                             AS descricaopedido,
  STR_TO_DATE(p.dataVisual, '%d-%m-%Y %H:%i:%s') AS dt_pedido,
  MONTH(p.dataVisual)                                   AS mes,
  YEAR(p.dataVisual)                                    AS ano,
  p.km                                                  AS km,
  su.id                                                 AS system_unit_id,
  ent.id                                                AS entidade_id,
  dep.id                                                AS departamento_unit_id,
  sus.id                                                AS system_users_id,
  tm.id                                                 AS tipo_manutencao_id,
  null as estado_pedido_frotas_id,
  p.data_finalizacao,
  p.data_aprovacao,
  p.status,
  p.enviar,
  p.motorista_id,
  p.motorista,
  p.usuario_id,
  p.usuario_aprovou,
  p.usuariocliente_id
  
FROM gestaonp3benefic_dbsystem_np3.pedidos p
JOIN gestaonp3benefic_dbgestao.veiculos vei
  ON vei.idold = p.veiculo_id
